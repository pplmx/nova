### 1. Create include/cuda/topology/topology_map.h

```cpp
#pragma once

#include <cuda_runtime.h>

#include <cstdint>
#include <map>
#include <optional>
#include <string>
#include <vector>

namespace cuda::topology {

enum class NicType {
    Unknown,
    Ethernet,
    InfiniBand,
    RoCE,
};

enum class PathType {
    IntraNode,  // Within same node (NVLink/PCIe)
    InterNode,  // Across nodes (network)
};

struct NicInfo {
    std::string name;
    std::string mac_address;
    NicType type = NicType::Unknown;
    uint64_t bandwidth_gbps = 0;
    bool is_rdma_capable = false;
    std::string ip_address;
};

struct TopologyMap {
    int node_id = 0;
    std::string hostname;
    std::vector<NicInfo> nics;
    
    PathType classify_path(int other_node_id, const std::string& other_hostname) const;
    
    NicInfo select_best_nic() const;
    
    bool has_rdma() const;
};

class TopologyDetector {
public:
    static TopologyMap detect();
    
    static std::string probe_nic_type(const std::string& interface);
    static uint64_t estimate_bandwidth(const std::string& interface);
    static bool is_rdma_capable(const std::string& interface);
};

void validate_topology(const TopologyMap& local, 
                       const std::vector<TopologyMap>& all_nodes);

}  // namespace cuda::topology
```

### 2. Create src/cuda/topology/topology_map.cpp

```cpp
#include "cuda/topology/topology_map.h"

#include <fstream>
#include <sstream>
#include <thread>

namespace cuda::topology {

namespace {

bool file_exists(const std::string& path) {
    std::ifstream f(path);
    return f.good();
}

std::string read_file(const std::string& path) {
    std::ifstream f(path);
    std::stringstream ss;
    ss << f.rdbuf();
    return ss.str();
}

std::string exec(const char* cmd) {
    std::array<char, 128> buffer;
    std::string result;
    std::unique_ptr<FILE, decltype(&pclose)> pipe(popen(cmd, "r"), pclose);
    while (fgets(buffer.data(), buffer.size(), pipe.get()) != nullptr) {
        result += buffer.data();
    }
    return result;
}

}  // namespace

TopologyMap TopologyDetector::detect() {
    TopologyMap map;
    
    char hostname[256];
    gethostname(hostname, sizeof(hostname));
    map.hostname = hostname;
    
    const std::string sys_class_net = "/sys/class/net";
    for (const auto& entry : std::filesystem::directory_iterator(sys_class_net)) {
        std::string name = entry.path().filename();
        if (name == "lo") continue;
        
        NicInfo nic;
        nic.name = name;
        
        const auto type_path = entry.path() / "type";
        if (file_exists(type_path.string())) {
            int type = std::stoi(read_file(type_path.string()));
            if (type == 32) nic.type = NicType::InfiniBand;
            else nic.type = NicType::Ethernet;
        }
        
        nic.is_rdma_capable = is_rdma_capable(name);
        nic.bandwidth_gbps = estimate_bandwidth(name);
        
        map.nics.push_back(nic);
    }
    
    return map;
}

std::string TopologyDetector::probe_nic_type(const std::string& interface) {
    const std::string ib_dev = "/sys/class/infiniband/";
    for (const auto& entry : std::filesystem::directory_iterator(ib_dev)) {
        std::string name = entry.path().filename();
        const auto ports_dir = entry.path() / "ports";
        if (std::filesystem::exists(ports_dir)) {
            for (const auto& port : std::filesystem::directory_iterator(ports_dir)) {
                const auto lid_path = port.path() / "lid";
                if (file_exists(lid_path.string())) {
                    return "InfiniBand";
                }
            }
        }
    }
    return "Ethernet";
}

uint64_t TopologyDetector::estimate_bandwidth(const std::string& interface) {
    const auto speed_path = "/sys/class/net/" + interface + "/speed";
    if (file_exists(speed_path)) {
        int speed = std::stoi(read_file(speed_path));
        return speed;
    }
    return 0;
}

bool TopologyDetector::is_rdma_capable(const std::string& interface) {
    const auto rdma_path = "/sys/class/net/" + interface + "/device/infiniband";
    return file_exists(rdma_path);
}

PathType TopologyMap::classify_path(int other_node_id, 
                                     const std::string& other_hostname) const {
    if (node_id == other_node_id || hostname == other_hostname) {
        return PathType::IntraNode;
    }
    return PathType::InterNode;
}

NicInfo TopologyMap::select_best_nic() const {
    NicInfo best;
    for (const auto& nic : nics) {
        if (nic.is_rdma_capable && nic.bandwidth_gbps > best.bandwidth_gbps) {
            best = nic;
        }
    }
    return nics.empty() ? NicInfo{} : nics[0];
}

bool TopologyMap::has_rdma() const {
    for (const auto& nic : nics) {
        if (nic.is_rdma_capable) return true;
    }
    return false;
}

}  // namespace cuda::topology
```
