```cpp
#pragma once

#include <cuda_runtime.h>
#include <string>
#include <vector>

namespace cuda::topology {

enum class CollectiveAlgorithm {
    Ring,      // Default, good for all topologies
    Tree,      // Better for large messages
    CollNet,   // Best for all-reduce with dedicated network
    Pipeline,  // For large transfers
};

struct AlgorithmConfig {
    CollectiveAlgorithm algorithm = CollectiveAlgorithm::Ring;
    int min_message_size = 0;
    int max_message_size = 1 << 30;
    std::string nccl_tuning_env;
};

class CollectiveSelector {
public:
    static AlgorithmConfig select_for_allreduce(
        uint64_t message_size_bytes,
        bool has_rdma,
        const std::string& network_type);
    
    static AlgorithmConfig select_for_broadcast(
        uint64_t message_size_bytes,
        int num_receivers);
    
    static AlgorithmConfig select_for_allgather(
        uint64_t message_size_bytes,
        int num_workers);
    
    static void set_nccl_env(const AlgorithmConfig& config);
    
    static AlgorithmConfig from_env();
};

class CollectiveProfiler {
public:
    struct BenchmarkResult {
        double bandwidth_gbps;
        double latency_us;
        CollectiveAlgorithm algorithm;
    };
    
    BenchmarkResult profile_allreduce(const void* send_buf,
                                      void* recv_buf,
                                      size_t count,
                                      cudaStream_t stream);
    
    void generate_report(const std::vector<BenchmarkResult>& results);
};

}  // namespace cuda::topology
```

### 2. Create src/cuda/topology/collective_selector.cpp

```cpp
#include "cuda/topology/collective_selector.h"

#include <algorithm>
#include <cstdlib>
#include <sstream>

namespace cuda::topology {

namespace {

constexpr int RDMA_THRESHOLD_BYTES = 1024 * 1024;  // 1MB
constexpr int TREE_THRESHOLD_BYTES = 32 * 1024 * 1024;  // 32MB
constexpr int COLLNET_THRESHOLD_BYTES = 64 * 1024 * 1024;  // 64MB

std::string algorithm_to_string(CollectiveAlgorithm algo) {
    switch (algo) {
        case CollectiveAlgorithm::Ring: return "Ring";
        case CollectiveAlgorithm::Tree: return "Tree";
        case CollectiveAlgorithm::CollNet: return "CollNet";
        case CollectiveAlgorithm::Pipeline: return "Pipeline";
        default: return "Unknown";
    }
}

}  // namespace

AlgorithmConfig CollectiveSelector::select_for_allreduce(
    uint64_t message_size_bytes,
    bool has_rdma,
    const std::string& network_type) {
    
    AlgorithmConfig config;
    
    if (has_rdma && network_type == "InfiniBand") {
        if (message_size_bytes >= COLLNET_THRESHOLD_BYTES) {
            config.algorithm = CollectiveAlgorithm::CollNet;
        } else if (message_size_bytes >= TREE_THRESHOLD_BYTES) {
            config.algorithm = CollectiveAlgorithm::Tree;
        } else {
            config.algorithm = CollectiveAlgorithm::Ring;
        }
    } else {
        if (message_size_bytes >= TREE_THRESHOLD_BYTES) {
            config.algorithm = CollectiveAlgorithm::Tree;
        } else {
            config.algorithm = CollectiveAlgorithm::Ring;
        }
    }
    
    return config;
}

AlgorithmConfig CollectiveSelector::select_for_broadcast(
    uint64_t message_size_bytes,
    int num_receivers) {
    
    AlgorithmConfig config;
    
    if (num_receivers > 8 && message_size_bytes > TREE_THRESHOLD_BYTES) {
        config.algorithm = CollectiveAlgorithm::Tree;
    } else {
        config.algorithm = CollectiveAlgorithm::Ring;
    }
    
    return config;
}

AlgorithmConfig CollectiveSelector::select_for_allgather(
    uint64_t message_size_bytes,
    int num_workers) {
    
    AlgorithmConfig config;
    
    if (message_size_bytes >= TREE_THRESHOLD_BYTES && num_workers > 4) {
        config.algorithm = CollectiveAlgorithm::Ring;
    } else {
        config.algorithm = CollectiveAlgorithm::Ring;
    }
    
    return config;
}

void CollectiveSelector::set_nccl_env(const AlgorithmConfig& config) {
    std::ostringstream env;
    env << "NCCL_ALGO=" << algorithm_to_string(config.algorithm);
    
    if (!config.nccl_tuning_env.empty()) {
        env << "," << config.nccl_tuning_env;
    }
    
    setenv("NCCL_TUNING", env.str().c_str(), 1);
}

AlgorithmConfig CollectiveSelector::from_env() {
    AlgorithmConfig config;
    
    const char* nccl_algo = getenv("NCCL_ALGO");
    if (nccl_algo) {
        if (strcmp(nccl_algo, "Ring") == 0) config.algorithm = CollectiveAlgorithm::Ring;
        else if (strcmp(nccl_algo, "Tree") == 0) config.algorithm = CollectiveAlgorithm::Tree;
        else if (strcmp(nccl_algo, "CollNet") == 0) config.algorithm = CollectiveAlgorithm::CollNet;
    }
    
    return config;
}

CollectiveProfiler::BenchmarkResult 
CollectiveProfiler::profile_allreduce(const void* send_buf,
                                      void* recv_buf,
                                      size_t count,
                                      cudaStream_t stream) {
    BenchmarkResult result;
    result.algorithm = CollectiveAlgorithm::Ring;
    
    cudaEvent_t start, stop;
    cudaEventCreate(&start);
    cudaEventCreate(&stop);
    
    cudaEventRecord(start, stream);
    cudaEventRecord(stop, stream);
    cudaEventSynchronize(stop);
    
    float ms;
    cudaEventElapsedTime(&ms, start, stop);
    
    result.latency_us = ms * 1000.0;
    result.bandwidth_gbps = (count * sizeof(float)) / (ms * 1e-3) / 1e9;
    
    cudaEventDestroy(start);
    cudaEventDestroy(stop);
    
    return result;
}

void CollectiveProfiler::generate_report(
    const std::vector<BenchmarkResult>& results) {
    
    std::printf("=== NCCL Collective Benchmark Report ===\n");
    for (const auto& r : results) {
        std::printf("Algorithm: %s, BW: %.2f GB/s, Latency: %.2f us\n",
                    algorithm_to_string(r.algorithm).c_str(),
                    r.bandwidth_gbps,
                    r.latency_us);
    }
}

}  // namespace cuda::topology
```
