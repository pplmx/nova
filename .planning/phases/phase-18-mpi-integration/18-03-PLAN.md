### 2. Add environment/config support in mpi_context.cpp

```cpp
void MpiContext::initialize(const MpiConfig& config) {
    if (initialized_) {
        return;
    }
    
    int provided;
    int required = config.thread_level;
    MPI_CHECK(MPI_Init_thread(nullptr, nullptr, required, &provided));
    
    // Fall back if requested level not available
    if (provided < required) {
        // Log warning
    }
    
    MPI_CHECK(MPI_Comm_rank(MPI_COMM_WORLD, &world_rank_));
    MPI_CHECK(MPI_Comm_size(MPI_COMM_WORLD, &world_size_));
    
    compute_local_rank();
    compute_node_id();
    initialized_ = true;
}
```

### 4. Create mpi_validation.h for startup checks

```cpp
// Validate MPI environment at startup
void validate_mpi_environment() {
    int initialized, finalized;
    MPI_Initialized(&initialized);
    MPI_Finalized(&finalized);
    
    if (finalized) {
        throw std::runtime_error("MPI already finalized");
    }
    
    if (!initialized) {
        throw std::runtime_error("MPI not initialized");
    }
}
```
