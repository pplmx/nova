```cpp
#pragma once

#include <cuda_runtime.h>
#include <mpi.h>

#include <cstdint>
#include <memory>
#include <stdexcept>
#include <string>

namespace cuda::mpi {

class MpiException : public std::runtime_error {
public:
    MpiException(int error_code, const char* file, int line);
    [[nodiscard]] int error_code() const noexcept { return error_code_; }
private:
    int error_code_;
};

#define MPI_CHECK(call) \
    do { \
        int _err = call; \
        if (_err != MPI_SUCCESS) { \
            throw cuda::mpi::MpiException(_err, __FILE__, __LINE__); \
        } \
    } while (0)

class MpiContext {
public:
    static MpiContext& instance();
    
    MpiContext();
    ~MpiContext();
    
    void initialize(int* argc, char*** argv);
    void finalize();
    
    [[nodiscard]] int world_rank() const { return world_rank_; }
    [[nodiscard]] int world_size() const { return world_size_; }
    [[nodiscard]] int local_rank() const { return local_rank_; }
    [[nodiscard]] int local_size() const { return local_size_; }
    [[nodiscard]] int node_id() const { return node_id_; }
    [[nodiscard]] bool is_main_process() const { return world_rank_ == 0; }
    [[nodiscard]] bool initialized() const { return initialized_; }
    
    int get_local_device_id() const;
    
private:
    void compute_local_rank();
    void compute_node_id();
    
    int world_rank_ = -1;
    int world_size_ = 1;
    int local_rank_ = 0;
    int local_size_ = 1;
    int node_id_ = 0;
    bool initialized_ = false;
};

}  // namespace cuda::mpi
```

### 2. Create src/cuda/mpi/mpi_context.cpp

```cpp
#include "cuda/mpi/mpi_context.h"

#include <algorithm>
#include <cstdlib>
#include <map>
#include <mutex>
#include <vector>

namespace cuda::mpi {

MpiException::MpiException(int error_code, const char* file, int line)
    : std::runtime_error("MPI error: " + std::to_string(error_code) +
                         " at " + std::string(file) + ":" + std::to_string(line)),
      error_code_(error_code) {}

MpiContext& MpiContext::instance() {
    static MpiContext ctx;
    return ctx;
}

MpiContext::MpiContext() = default;

MpiContext::~MpiContext() {
    if (initialized_) {
        MPI_CHECK(MPI_Finalize());
    }
}

void MpiContext::initialize(int* argc, char*** argv) {
    if (initialized_) {
        return;
    }
    
    int provided;
    MPI_CHECK(MPI_Init_thread(argc, argv, MPI_THREAD_MULTIPLE, &provided));
    
    MPI_CHECK(MPI_Comm_rank(MPI_COMM_WORLD, &world_rank_));
    MPI_CHECK(MPI_Comm_size(MPI_COMM_WORLD, &world_size_));
    
    compute_local_rank();
    compute_node_id();
    
    initialized_ = true;
}

void MpiContext::finalize() {
    if (initialized_) {
        MPI_CHECK(MPI_Finalize());
        initialized_ = false;
    }
}

void MpiContext::compute_local_rank() {
    char hostname[MPI_MAX_PROCESSOR_NAME];
    int len;
    MPI_CHECK(MPI_Get_processor_name(hostname, &len));
    
    std::vector<int> ranks(world_size_);
    std::vector<int> displs(world_size_);
    std::vector<char> names(MPI_MAX_PROCESSOR_NAME * world_size_);
    
    for (int i = 0; i < world_size_; ++i) {
        displs[i] = i * MPI_MAX_PROCESSOR_NAME;
    }
    
    MPI_CHECK(MPI_Allgather(hostname, MPI_MAX_PROCESSOR_NAME, MPI_CHAR,
                            names.data(), MPI_MAX_PROCESSOR_NAME, MPI_CHAR,
                            MPI_COMM_WORLD));
    
    std::map<std::string, int> node_rank_count;
    std::string my_hostname(hostname);
    
    for (int i = 0; i < world_size_; ++i) {
        std::string other_hostname(&names[i * MPI_MAX_PROCESSOR_NAME]);
        node_rank_count[other_hostname]++;
    }
    
    local_rank_ = 0;
    for (int i = 0; i < world_rank_; ++i) {
        std::string other_hostname(&names[i * MPI_MAX_PROCESSOR_NAME]);
        if (other_hostname == my_hostname) {
            local_rank_++;
        }
    }
}

int MpiContext::get_local_device_id() const {
    const char* cuda_visible = std::getenv("CUDA_VISIBLE_DEVICES");
    if (cuda_visible) {
        return local_rank_;
    }
    return local_rank_ % cudaGetDeviceCount();
}

}  // namespace cuda::mpi
```
