## Objective

## Implementation

### 1. Create cmake/FindMPI.cmake

```cmake
# FindMPI.cmake - Locate MPI library and headers
#
# Variables:
#   MPI_FOUND          - True if MPI is found
#   MPI_INCLUDE_DIRS   - Include directories
#   MPI_LIBRARIES      - Libraries to link
#   MPI_VERSION        - Version string
#
# Supports: OpenMPI, MPICH, Intel MPI

# Minimum version (MPI 3.1+ for MPI_T_* functions)
set(MPI_MIN_VERSION "3.1")

# Search paths
set(MPI_SEARCH_PATHS
    ${MPI_DIR}
    $ENV{MPI_DIR}
    /usr/lib/x86_64-linux-gnu
    /usr/lib64
    /usr/local/lib
)

# Find MPI header
find_path(MPI_INCLUDE_DIR
    NAMES mpi.h
    HINTS ${MPI_DIR} $ENV{MPI_DIR}
    PATHS ${MPI_SEARCH_PATHS}
    PATH_SUFFIXES include include/mpi
)

# Find MPI library
find_library(MPI_CXX_LIBRARY
    NAMES mpi_cxx mpi
    PATHS ${MPI_SEARCH_PATHS}
    PATH_SUFFIXES lib lib64
)

find_library(MPI_LIBRARY
    NAMES mpi
    PATHS ${MPI_SEARCH_PATHS}
    PATH_SUFFIXES lib lib64
)

# Version detection from mpi.h
if(MPI_INCLUDE_DIR AND EXISTS "${MPI_INCLUDE_DIR}/mpi.h")
    file(STRINGS "${MPI_INCLUDE_DIR}/mpi.h" MPI_VERSION_DEF
        REGEX "^#define[ \t]+MPI_VERSION[ \t]+[0-9]+")
    if(MPI_VERSION_DEF)
        string(REGEX REPLACE "^#define[ \t]+MPI_VERSION[ \t]+([0-9]+).*" "\\1"
            MPI_VERSION_MAJOR "${MPI_VERSION_DEF}")
    endif()
    
    file(STRINGS "${MPI_INCLUDE_DIR}/mpi.h" MPI_SUBVERSION_DEF
        REGEX "^#define[ \t]+MPI_SUBVERSION[ \t]+[0-9]+")
    if(MPI_SUBVERSION_DEF)
        string(REGEX REPLACE "^#define[ \t]+MPI_SUBVERSION[ \t]+([0-9]+).*" "\\1"
            MPI_VERSION_MINOR "${MPI_SUBVERSION_DEF}")
    endif()
    
    set(MPI_VERSION "${MPI_VERSION_MAJOR}.${MPI_VERSION_MINOR}")
endif()

# Create imported target
if(MPI_FOUND AND NOT TARGET MPI::MPI_CXX)
    add_library(MPI::MPI_CXX UNKNOWN IMPORTED)
    set_target_properties(MPI::MPI_CXX PROPERTIES
        IMPORTED_LOCATION "${MPI_CXX_LIBRARY}"
        INTERFACE_INCLUDE_DIRECTORIES "${MPI_INCLUDE_DIR}"
    )
endif()
```

### 2. Update CMakeLists.txt

```cmake
option(NOVA_ENABLE_MPI "Enable multi-node support via MPI" OFF)

if(NOVA_ENABLE_MPI)
    find_package(MPI 3.1 REQUIRED)
    
    # Add compile definitions
    target_compile_definitions(nova PUBLIC NOVA_MPI_ENABLED=1)
    
    # Add include directories
    target_include_directories(nova SYSTEM PUBLIC ${MPI_INCLUDE_DIRS})
    
    # Link libraries
    target_link_libraries(nova PUBLIC MPI::MPI_CXX)
endif()
```
