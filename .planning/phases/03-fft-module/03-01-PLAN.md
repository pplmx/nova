2. **Plan Creation Helpers**:

       void FFTPlan::create_plan_1d(size_t size, Precision precision, Direction direction) {
           int n = static_cast<int>(size);
           
           switch (precision) {
               case Precision::Single:
                   CUDA_CHECK(cufftPlan1d(&plan_, n, CUFFT_R2C, 1));
                   break;
               case Precision::Double:
                   CUDA_CHECK(cufftPlan1d(&plan_, n, CUFFT_D2Z, 1));
                   break;
               case Precision::ComplexToReal:
                   CUDA_CHECK(cufftPlan1d(&plan_, n, CUFFT_C2R, 1));
                   break;
               case Precision::ComplexToComplex:
                   CUDA_CHECK(cufftPlan1d(&plan_, n, CUFFT_C2C, 1));
                   break;
           }
       }
       
       void FFTPlan::create_plan_2d(size_t nx, size_t ny, Precision precision, Direction direction) {
           switch (precision) {
               case Precision::Single:
                   CUDA_CHECK(cufftPlan2d(&plan_, nx, ny, CUFFT_R2C));
                   break;
               case Precision::Double:
                   CUDA_CHECK(cufftPlan2d(&plan_, nx, ny, CUFFT_D2Z));
                   break;
               default:
                   throw std::invalid_argument("Unsupported precision for 2D transform");
           }
       }
       
       void FFTPlan::create_plan_3d(size_t nx, size_t ny, size_t nz, Precision precision, Direction direction) {
           switch (precision) {
               case Precision::Single:
                   CUDA_CHECK(cufftPlan3d(&plan_, nx, ny, nz, CUFFT_R2C));
                   break;
               case Precision::Double:
                   CUDA_CHECK(cufftPlan3d(&plan_, nx, ny, nz, CUFFT_D2Z));
                   break;
               default:
                   throw std::invalid_argument("Unsupported precision for 3D transform");
           }
       }
       
       void FFTPlan::destroy_plan() {
           if (plan_ != 0 && owns_plan_) {
               CUDA_CHECK(cufftDestroy(plan_));
               plan_ = 0;
           }
       }

           if (stream) {

           }
           CUDA_CHECK(cufftExecR2C(plan_, const_cast<float*>(input), output));

       void FFTPlan::inverse(const cuComplex* input, float* output, cudaStream_t stream) {
           if (stream) {

           CUDA_CHECK(cufftExecC2R(plan_, const_cast<cuComplex*>(input), output));
       }
       
       void FFTPlan::inverse(const cuDoubleComplex* input, double* output, cudaStream_t stream) {
           if (stream) {
               CUDA_CHECK(cufftSetStream(plan_, stream));

       }
       
       // Complex to complex transforms
       void FFTPlan::transform(Direction direction, const cuComplex* input,
                               cuComplex* output, cudaStream_t stream) {
           if (stream) {
               CUDA_CHECK(cufftSetStream(plan_, stream));

           auto dir = (direction == Direction::Forward) ? CUFFT_FORWARD : CUFFT_INVERSE;
           CUDA_CHECK(cufftExecC2C(plan_, const_cast<cuComplex*>(input), output, dir));
