#include <gtest/gtest.h>
#include <cuda/sparse/sparse_matrix.hpp>
#include <cuda/sparse/hyb_matrix.hpp>
#include <vector>
#include <cmath>

namespace nova {
namespace sparse {
namespace test {

// Helper to create a test CSR matrix
template<typename T>
SparseMatrixCSR<T> create_test_csr() {
    // 4x4 matrix:
    // [1, 0, 2, 0]
    // [0, 3, 0, 0]
    // [4, 0, 5, 6]
    // [0, 0, 7, 0]
    std::vector<T> values = {T{1}, T{2}, T{3}, T{4}, T{5}, T{6}, T{7}};
    std::vector<int> row_offsets = {0, 2, 3, 6, 7};
    std::vector<int> col_indices = {0, 2, 1, 0, 2, 3, 2};
    return SparseMatrixCSR<T>(std::move(values), std::move(row_offsets),
                               std::move(col_indices), 4, 4);
}

TEST(HYBConstruction, FromCSRBasic) {
    auto csr = create_test_csr<double>();
    auto hyb = SparseMatrixHYB<double>::FromCSR(csr);
    
    EXPECT_EQ(hyb.num_rows(), 4);
    EXPECT_EQ(hyb.num_cols(), 4);
    EXPECT_EQ(hyb.nnz(), 7);
}

TEST(HYBConstruction, Partitioning) {
    auto csr = create_test_csr<double>();
    auto hyb = SparseMatrixHYB<double>::FromCSR(csr, 2);
    
    // Row 0 has 2 nonzeros, Row 2 has 3 nonzeros (should go to ELL if threshold < 2)
    // Row 1 has 1 nonzero, Row 3 has 1 nonzero (should go to COO)
    int ell_count = hyb.ell_row_count();
    int coo_count = hyb.coo_row_count();
    
    EXPECT_EQ(ell_count + coo_count, 4);
    EXPECT_GT(ell_count, 0);
    EXPECT_GT(coo_count, 0);
}

TEST(HYBConstruction, FromEmptyCSR) {
    std::vector<double> values;
    std::vector<int> row_offsets = {0, 0, 0, 0};
    std::vector<int> col_indices;
    auto csr = SparseMatrixCSR<double>(std::move(values), std::move(row_offsets),
                                        std::move(col_indices), 3, 3);
    auto hyb = SparseMatrixHYB<double>::FromCSR(csr);
    
    EXPECT_EQ(hyb.num_rows(), 3);
    EXPECT_EQ(hyb.nnz(), 0);
}

TEST(HYBSpMV, Correctness) {
    auto csr = create_test_csr<double>();
    auto hyb = SparseMatrixHYB<double>::FromCSR(csr);
    
    double x[] = {1.0, 2.0, 3.0, 4.0};
    double y_csr[] = {0.0, 0.0, 0.0, 0.0};
    double y_hyb[] = {0.0, 0.0, 0.0, 0.0};
    
    // Reference CSR SpMV
    for (int i = 0; i < 4; ++i) {
        for (int j = csr.row_offsets()[i]; j < csr.row_offsets()[i + 1]; ++j) {
            y_csr[i] += csr.values()[j] * x[csr.col_indices()[j]];
        }
    }
    
    // HYB SpMV
    sparse_mv(hyb, x, y_hyb);
    
    for (int i = 0; i < 4; ++i) {
        EXPECT_NEAR(y_csr[i], y_hyb[i], 1e-10);
    }
}

TEST(HYBSpMV, IrregularMatrix) {
    // Create matrix with highly irregular row distribution
    // Row 0: 10 nonzeros (dense-ish)
    // Row 1: 1 nonzero
    // Row 2: 8 nonzeros
    // Row 3: 1 nonzero
    // Row 4: 9 nonzeros
    std::vector<double> values;
    std::vector<int> row_offsets = {0};
    std::vector<int> col_indices;
    
    for (int row = 0; row < 5; ++row) {
        int nnz = (row % 2 == 0) ? (10 - row) : 1;  // 10, 1, 8, 1, 9
        for (int j = 0; j < nnz; ++j) {
            values.push_back(static_cast<double>(row * 10 + j));
            col_indices.push_back(j);
        }
        row_offsets.push_back(static_cast<int>(values.size()));
    }
    
    auto csr = SparseMatrixCSR<double>(std::move(values), std::move(row_offsets),
                                        std::move(col_indices), 5, 20);
    auto hyb = SparseMatrixHYB<double>::FromCSR(csr);
    
    // Verify partition
    EXPECT_EQ(hyb.ell_row_count() + hyb.coo_row_count(), 5);
    
    // Verify SpMV correctness
    std::vector<double> x(20, 1.0);
    std::vector<double> y_csr(5, 0.0), y_hyb(5, 0.0);
    
    for (int i = 0; i < 5; ++i) {
        for (int j = csr.row_offsets()[i]; j < csr.row_offsets()[i + 1]; ++j) {
            y_csr[i] += csr.values()[j] * x[csr.col_indices()[j]];
        }
    }
    
    sparse_mv(hyb, x.data(), y_hyb.data());
    
    for (int i = 0; i < 5; ++i) {
        EXPECT_NEAR(y_csr[i], y_hyb[i], 1e-10);
    }
}

} // namespace test
} // namespace sparse
} // namespace nova

#include <cuda/sparse/roofline.hpp>

namespace nova {
namespace sparse {
namespace test {

// Tests for RooflineAnalyzer enhancements

TEST(RooflineClassification, ComputeBoundDetection) {
    RooflineAnalyzer analyzer;
    
    // High AI -> compute bound
    // For FP32 on typical GPU: peak ~20000 GFLOP/s, BW ~2000 GB/s
    // AI > 10 means bandwidth ceiling > 20000 -> compute bound
    auto result = analyzer.classify_with_confidence(15.0, 20000.0);
    
    EXPECT_EQ(result.bound, PerformanceBound::COMPUTE_BOUND);
    EXPECT_GT(result.confidence_percent, 50.0);
}

TEST(RooflineClassification, MemoryBoundDetection) {
    RooflineAnalyzer analyzer;
    
    // Low AI -> memory bound
    // AI = 0.5 means bandwidth ceiling = 0.5 * 2000 = 1000 < 20000
    auto result = analyzer.classify_with_confidence(0.5, 20000.0);
    
    EXPECT_EQ(result.bound, PerformanceBound::MEMORY_BOUND);
    EXPECT_GT(result.confidence_percent, 50.0);
}

TEST(RooflineClassification, BalancedDetection) {
    RooflineAnalyzer analyzer;
    
    // Balanced: bandwidth ceiling close to compute ceiling
    // AI = 10 on typical GPU: 10 * 2000 = 20000 ≈ 20000
    auto result = analyzer.classify_with_confidence(10.0, 20000.0);
    
    EXPECT_EQ(result.bound, PerformanceBound::BALANCED);
}

TEST(RooflineJsonExport, SingleKernel) {
    RooflineAnalyzer analyzer;
    RooflineMetrics m;
    m.kernel_name = "spmv_test";
    m.arithmetic_intensity = 1.5;
    m.achieved_gflops = 500.0;
    m.peak_gflops = 10000.0;
    m.achieved_bandwidth_gbps = 333.0;
    m.peak_bandwidth_gbps = 1000.0;
    m.bound = PerformanceBound::MEMORY_BOUND;
    
    std::string json = analyzer.to_json(m);
    
    // Verify JSON structure
    EXPECT_NE(json.find("\"kernel_name\": \"spmv_test\""), std::string::npos);
    EXPECT_NE(json.find("\"arithmetic_intensity\": 1.5"), std::string::npos);
    EXPECT_NE(json.find("\"performance_bound\": \"MEMORY_BOUND\""), std::string::npos);
}

TEST(RooflineJsonExport, DeviceInfo) {
    RooflineAnalyzer analyzer;
    std::string json = analyzer.to_json_device_info();
    
    EXPECT_NE(json.find("\"peak_gflops\""), std::string::npos);
    EXPECT_NE(json.find("\"peak_bandwidth_gbps\""), std::string::npos);
    EXPECT_NE(json.find("\"compute_capability\""), std::string::npos);
}

TEST(RooflineJsonExport, MultiKernelComparison) {
    RooflineAnalysis analysis;
    
    // Add kernel 1
    RooflineMetrics m1;
    m1.kernel_name = "csr_spmv";
    m1.arithmetic_intensity = 0.8;
    m1.achieved_gflops = 400.0;
    m1.peak_gflops = 10000.0;
    m1.achieved_bandwidth_gbps = 500.0;
    m1.peak_bandwidth_gbps = 1000.0;
    m1.bound = PerformanceBound::MEMORY_BOUND;
    analysis.add_kernel(m1);
    
    // Add kernel 2
    RooflineMetrics m2;
    m2.kernel_name = "hyb_spmv";
    m2.arithmetic_intensity = 1.2;
    m2.achieved_gflops = 600.0;
    m2.peak_gflops = 10000.0;
    m2.achieved_bandwidth_gbps = 500.0;
    m2.peak_bandwidth_gbps = 1000.0;
    m2.bound = PerformanceBound::MEMORY_BOUND;
    analysis.add_kernel(m2);
    
    DevicePeaks peaks;
    peaks.fp32_peak_gflops = 10000.0;
    peaks.memory_bandwidth_gbps = 1000.0;
    peaks.compute_capability_major = 8;
    peaks.compute_capability_minor = 0;
    analysis.set_device_info(peaks, Precision::FP32, "Test GPU");
    
    std::string json = analysis.to_json();
    
    // Verify JSON structure
    EXPECT_NE(json.find("\"metadata\""), std::string::npos);
    EXPECT_NE(json.find("\"device_name\": \"Test GPU\""), std::string::npos);
    EXPECT_NE(json.find("\"device_peaks\""), std::string::npos);
    EXPECT_NE(json.find("\"kernels\""), std::string::npos);
    EXPECT_NE(json.find("\"csr_spmv\""), std::string::npos);
    EXPECT_NE(json.find("\"hyb_spmv\""), std::string::npos);
    
    // Verify array structure (two kernel objects)
    EXPECT_NE(json.find("{\"name\": \"csr_spmv\""), std::string::npos);
    EXPECT_NE(json.find("{\"name\": \"hyb_spmv\""), std::string::npos);
}

TEST(RooflineJsonExport, ValidJsonSyntax) {
    RooflineAnalyzer analyzer;
    RooflineMetrics m;
    m.kernel_name = "test";
    m.arithmetic_intensity = 1.0;
    m.achieved_gflops = 100.0;
    m.peak_gflops = 1000.0;
    m.bound = PerformanceBound::BALANCED;
    
    std::string json = analyzer.to_json(m);
    
    // Basic JSON validity check
    EXPECT_EQ(json.front(), '{');
    EXPECT_EQ(json.back(), '}');
    
    // Count matching braces
    size_t open = std::count(json.begin(), json.end(), '{');
    size_t close = std::count(json.begin(), json.end(), '}');
    EXPECT_EQ(open, close);
}

} // namespace test
} // namespace sparse
} // namespace nova

    ::testing::InitGoogleTest(&argc, argv);
    return RUN_ALL_TESTS();
}
