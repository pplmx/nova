## Implementation

### Create `include/cuda/error/cuda_error.hpp`

```cpp
namespace nova::error {

// Error codes mirroring CUDA error types
enum class cuda_error_code {
    success = 0,
    missing_configuration = 1,
    // ... all cudaError_t values
};

// Error category for std::error_code integration
class cuda_error_category : public std::error_category {
public:
    const char* name() const noexcept override;
    std::string message(int ev) const override;
    
    // Recovery hint mapping
    std::string recovery_hint(cuda_error_code code) const;
};

// Global category instance
const std::error_category& cuda_category();

// Convenience make_error_code
std::error_code make_error_code(cudaError_t err, const char* operation = nullptr,
                                 int device = -1, intptr_t stream = 0);

// Error info struct with full context
struct cuda_error_info {
    cudaError_t code;
    const char* operation;  // e.g., "cudaMalloc", "ReduceKernel"
    const char* file;
    int line;
    int device_id;
    intptr_t stream;
    std::string message;    // Formatted full message
    
    std::string_view recovery_hint() const;
};

// RAII guard for scoped error capture
class cuda_error_guard {
public:
    cuda_error_guard(const char* operation, int device = -1);
    ~cuda_error_guard();
    
    void check(cudaError_t err);  // Throws on error
    
    cuda_error_info info() const { return info_; }
    
private:
    cuda_error_info info_;
    bool ok_ = true;
};

// Recovery hint mapping (internal)
const char* get_recovery_hint(cudaError_t err);

} // namespace nova::error
```

```cpp
// New improved macros (to replace 201 usages)
#define NOVA_CHECK(call) \
    nova::error::cuda_error_guard{#call, -1}.check(call)

// With explicit context
#define NOVA_CHECK_CTX(call, device, stream) \
    nova::error::cuda_error_guard{#call, device, stream}.check(call)
```
