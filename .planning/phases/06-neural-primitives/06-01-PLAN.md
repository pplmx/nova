       void matmul(
           const float* A,
           const float* B,
           float* C,
           int m, int n, int k,
           float alpha, float beta,
           cublasHandle_t handle,
           cudaStream_t stream
       ) {
           // Get or create cuBLAS handle
           int current_device;
           CUDA_CHECK(cudaGetDevice(&current_device));
           if (!handle) {
               handle = get_cublas_handle(current_device);
           }

               CUBLAS_CHECK(cublasSetStream(handle, stream));
           }
           
           // cuBLAS uses column-major order
           // For row-major: C^T = B^T * A^T
           // So we compute: C^T = (B^T) * (A^T)
           // Where A (MxK) -> A^T (KxM), B (KxN) -> B^T (NxK), C (MxN) -> C^T (NxM)
           
           const float* d_A = A;
           const float* d_B = B;
           float* d_C = C;
           
           // Note: For true row-major semantics, we transpose:
           // C = A * B in row-major means C^T = B^T * A^T in column-major
           // So we call: cublasSgemm(B^T, A^T, ...)
           
           // Leading dimensions for row-major storage
           // A is MxK, stored as M rows of K elements each
           // In row-major: lda = K (stride between consecutive rows)
           int lda = k;  // Columns of A
           int ldb = n;  // Columns of B  
           int ldc = n;  // Columns of C
           
           // For row-major A(MxK) * B(KxN) = C(MxN)
           // Need to treat A as (KxM)^T and B as (N x K)^T
           // This means we compute C^T = B^T * A^T
           
           cublasOperation_t transa = CUBLAS_OP_T;  // A -> A^T
           cublasOperation_t transb = CUBLAS_OP_T;  // B -> B^T
           
           // Actually for row-major: C = A * B means we compute:
           // C^T = B^T * A^T with cublas
           // transa = CUBLAS_OP_T (use A^T), transb = CUBLAS_OP_T (use B^T)
           
           // NN-01: Matrix multiply matches cuBLAS reference within 1e-3
           CUBLAS_CHECK(cublasSgemm(
               handle,
               transb, transa,  // B^T * A^T = C^T
               n, m, k,         // Op(B)=NxK, Op(A)=KxM, so result is NxM

   namespace cuda::neural {
       // CUDA kernel for computing max value per row (for numerical stability)
       // NN-02: Uses log-sum-exp trick to prevent overflow
       // softmax(x_i) = exp(x_i - max) / sum(exp(x_j - max))
       __global__ void softmax_max_kernel(
           const float* __restrict__ input,
           float* __restrict__ max_vals,
           int rows,
           int cols
       ) {
           extern __shared__ float sdata[];
           int row = blockIdx.x;
           int tid = threadIdx.x;
           
           if (row >= rows) return;
           
           // Load row into shared memory and find max
           float val = (tid < cols) ? input[row * cols + tid] : -INFINITY;
           sdata[tid] = val;
           __syncthreads();
           
           // Parallel reduction for max
           for (int s = blockDim.x / 2; s > 0; s >>= 1) {
               if (tid < s && tid + s < cols) {
                   sdata[tid] = fmaxf(sdata[tid], sdata[tid + s]);
               }
               __syncthreads();
           }
           
           if (tid == 0) {
               max_vals[row] = sdata[0];
           }
       }
       
       // CUDA kernel for softmax computation (subtract max, exp, sum)
       __global__ void softmax_exp_sum_kernel(
           const float* __restrict__ input,
           const float* __restrict__ max_vals,
           float* __restrict__ exp_vals,
           int rows,
           int cols
       ) {
           int row = blockIdx.x;
           int col = threadIdx.x;
           
           if (row >= rows || col >= cols) return;
           
           float x = input[row * cols + col];
           float max_val = max_vals[row];
           
           // Subtract max for numerical stability (log-sum-exp trick)
           float exp_val = expf(x - max_val);
           exp_vals[row * cols + col] = exp_val;
       }
       
       // CUDA kernel for computing sum of exp values per row
       __global__ void softmax_sum_kernel(
           const float* __restrict__ exp_vals,
           float* __restrict__ sums,
           int rows,
           int cols
       ) {
           extern __shared__ float sdata[];
           int row = blockIdx.x;
           int tid = threadIdx.x;
           
           if (row >= rows) return;
           
           float val = (tid < cols) ? exp_vals[row * cols + tid] : 0.0f;
           sdata[tid] = val;
           __syncthreads();
           
           // Parallel reduction for sum
           for (int s = blockDim.x / 2; s > 0; s >>= 1) {
               if (tid < s && tid + s < cols) {
                   sdata[tid] += sdata[tid + s];
               }
               __syncthreads();
           }
           
           if (tid == 0) {
               sums[row] = sdata[0];
           }
       }
       
       // CUDA kernel for final normalization
       __global__ void softmax_normalize_kernel(
           const float* __restrict__ exp_vals,
           const float* __restrict__ sums,
           float* __restrict__ output,
           int rows,
           int cols
       ) {
           int row = blockIdx.x;
           int col = threadIdx.x;
           
           if (row >= rows || col >= cols) return;
           
           float exp_val = exp_vals[row * cols + col];
           float sum = sums[row];
           
           output[row * cols + col] = exp_val / sum;
       }
   }

   namespace cuda::neural {
       // Kernel to compute mean per row
       __global__ void layer_norm_mean_kernel(
           const float* __restrict__ input,
           float* __restrict__ mean,
           int batch_size,
           int feature_size
       ) {
           extern __shared__ float sdata[];
           int batch = blockIdx.x;
           int tid = threadIdx.x;
           
           if (batch >= batch_size) return;
           
           // Compute sum
           float sum = 0.0f;
           for (int i = tid; i < feature_size; i += blockDim.x) {
               sum += input[batch * feature_size + i];
           }
           sdata[tid] = sum;
           __syncthreads();
           
           // Reduce sum
           for (int s = blockDim.x / 2; s > 0; s >>= 1) {
               if (tid < s) {
                   sdata[tid] += sdata[tid + s];
               }
               __syncthreads();
           }
           
           if (tid == 0) {
               mean[batch] = sdata[0] / feature_size;
           }
       }
       
       // Kernel to compute variance per row
       __global__ void layer_norm_var_kernel(
           const float* __restrict__ input,
           const float* __restrict__ mean,
           float* __restrict__ variance,
           int batch_size,
           int feature_size,
           float eps
       ) {
           extern __shared__ float sdata[];
           int batch = blockIdx.x;
           int tid = threadIdx.x;
           
           if (batch >= batch_size) return;
           
           float m = mean[batch];
           float sum_sq = 0.0f;
           
           for (int i = tid; i < feature_size; i += blockDim.x) {
               float diff = input[batch * feature_size + i] - m;
               sum_sq += diff * diff;
           }
           sdata[tid] = sum_sq;
           __syncthreads();
           
           // Reduce sum of squares
           for (int s = blockDim.x / 2; s > 0; s >>= 1) {
               if (tid < s) {
                   sdata[tid] += sdata[tid + s];
               }
               __syncthreads();
           }
           
           if (tid == 0) {
               variance[batch] = sdata[0] / feature_size + eps;
           }
       }
       
       // Kernel to normalize and apply gamma/beta
       __global__ void layer_norm_normalize_kernel(
           const float* __restrict__ input,
           const float* __restrict__ mean,
           const float* __restrict__ variance,
           const float* __restrict__ gamma,
           const float* __restrict__ beta,
           float* __restrict__ output,
           int batch_size,
           int feature_size,
           float eps
       ) {
           int batch = blockIdx.x;
           int feature = threadIdx.x;
           
           if (batch >= batch_size || feature >= feature_size) return;
           
           int idx = batch * feature_size + feature;
           
           float m = mean[batch];
           float var = variance[batch];
           float std = sqrtf(var);
           
           // Normalize: (x - mean) / sqrt(var + eps)
           float normalized = (input[idx] - m) / std;
           
           // Scale and shift: gamma * normalized + beta
           float g = gamma ? gamma[feature] : 1.0f;
           float b = beta ? beta[feature] : 0.0f;
           
           output[idx] = g * normalized + b;
       }
   }

           CUDA_CHECK(cudaMalloc(&d_mean, batch_size * sizeof(float)));
           CUDA_CHECK(cudaMalloc(&d_var, batch_size * sizeof(float)));
           CUDA_CHECK(cudaMalloc(&d_gamma, feature_size * sizeof(float)));

               CUDA_CHECK(cudaMemcpy(d_gamma, gamma, feature_size * sizeof(float), cudaMemcpyHostToDevice));
           } else {

           }

           // NN-04: Produces correct mean (approx 0) after normalization

                   beta ? d_beta : nullptr,

           CUDA_CHECK(cudaMemcpy(variance, d_var, batch_size * sizeof(float), cudaMemcpyDeviceToHost));
           
           // Cleanup
           cudaFree(d_input);
           cudaFree(d_mean);
           cudaFree(d_var);
           cudaFree(d_gamma);
           cudaFree(d_beta);
