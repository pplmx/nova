### 2. Create src/cuda/multinode/hierarchical_collectives.cpp

```cpp
#include "cuda/multinode/hierarchical_collectives.h"

#include "cuda/common/define.h"

namespace cuda::multinode {

HierarchicalAllReduce::HierarchicalAllReduce(ncclComm_t local_comm,
                                             ncclComm_t global_comm,
                                             cudaStream_t local_stream,
                                             cudaStream_t global_stream)
    : local_comm_(local_comm),
      global_comm_(global_comm),
      local_stream_(local_stream),
      global_stream_(global_stream) {}

void HierarchicalAllReduce::all_reduce(const void* send_buf,
                                       void* recv_buf,
                                       size_t count,
                                       ncclDataType_t dtype,
                                       ncclRedOp_t op) {
#if NOVA_NCCL_ENABLED
    NCCL_CHECK(ncclAllReduce(send_buf, recv_buf, count, dtype, op,
                             local_comm_, local_stream_));
    
    cudaStreamSynchronize(local_stream_);
    
    NCCL_CHECK(ncclAllReduce(recv_buf, recv_buf, count, dtype, op,
                             global_comm_, global_stream_));
#else
    (void)send_buf;
    (void)recv_buf;
    (void)count;
    (void)dtype;
    (void)op;
#endif
}

void HierarchicalAllReduce::synchronize() {
#if NOVA_NCCL_ENABLED
    cudaStreamSynchronize(global_stream_);
#endif
}

HierarchicalBroadcast::HierarchicalBroadcast(ncclComm_t local_comm,
                                             ncclComm_t global_comm,
                                             cudaStream_t stream)
    : local_comm_(local_comm),
      global_comm_(global_comm),
      stream_(stream) {}

void HierarchicalBroadcast::broadcast(const void* send_buf,
                                      void* recv_buf,
                                      size_t count,
                                      ncclDataType_t dtype,
                                      int root_rank,
                                      bool is_global_root) {
#if NOVA_NCCL_ENABLED
    if (is_global_root) {
        NCCL_CHECK(ncclBroadcast(send_buf, recv_buf, count, dtype, 0,
                                 global_comm_, stream_));
    }
    
    NCCL_CHECK(ncclBroadcast(send_buf, recv_buf, count, dtype, root_rank,
                             local_comm_, stream_));
#else
    (void)send_buf;
    (void)recv_buf;
    (void)count;
    (void)dtype;
    (void)root_rank;
    (void)is_global_root;
#endif
}

void HierarchicalBroadcast::synchronize() {
#if NOVA_NCCL_ENABLED
    cudaStreamSynchronize(stream_);
#endif
}

HierarchicalBarrier::HierarchicalBarrier(ncclComm_t local_comm,
                                         ncclComm_t global_comm,
                                         cudaStream_t stream)
    : local_comm_(local_comm),
      global_comm_(global_comm),
      stream_(stream) {}

void HierarchicalBarrier::wait() {
#if NOVA_NCCL_ENABLED
    NCCL_CHECK(ncclBarrier(local_comm_, stream_));
    cudaStreamSynchronize(stream_);
    NCCL_CHECK(ncclBarrier(global_comm_, stream_));
#endif
}

void HierarchicalBarrier::synchronize() {
#if NOVA_NCCL_ENABLED
    cudaStreamSynchronize(stream_);
#endif
}

}  // namespace cuda::multinode
```
