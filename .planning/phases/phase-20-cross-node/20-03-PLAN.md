### 1. Add fallback support in multi_node_context.cpp

```cpp
void MultiNodeContext::initialize(const NodeConfig& config) {
    // ... existing initialization ...
    
    if (!NCCL_ENABLED) {
        if (config.world_size > 1) {
            std::fprintf(stderr, "Warning: Multi-node requested but NCCL not available. "
                    "Falling back to single-node mode.\n");
        }
        global_size_ = 1;
        num_nodes_ = 1;
    }
    
    initialized_ = true;
}
```

### 2. Create tests/test_multinode.cpp

```cpp
#include <gtest/gtest.h>
#include "cuda/multinode/multi_node_context.h"
#include "cuda/multinode/hierarchical_collectives.h"

using namespace cuda::multinode;

class MultiNodeTest : public ::testing::Test {
protected:
    void SetUp() override {
    }
};

TEST_F(MultiNodeTest, SingleNodeFallback) {
    MultiNodeContext::NodeConfig config;
    config.world_rank = 0;
    config.world_size = 1;
    config.local_rank = 0;
    config.local_size = 1;
    config.node_id = 0;
    config.hostname = "localhost";
    
    auto& ctx = MultiNodeContext::instance();
    ctx.initialize(config);
    
    EXPECT_TRUE(ctx.is_initialized());
    EXPECT_EQ(ctx.global_size(), 1);
    EXPECT_EQ(ctx.num_nodes(), 1);
}

TEST_F(MultiNodeTest, HierarchicalAllReduce) {
#if NOVA_NCCL_ENABLED && NOVA_MPI_ENABLED
    // Skip actual test if not in multi-node environment
    GTEST_SKIP() << "Multi-node test requires MPI environment";
#else
    GTEST_SKIP() << "Multi-node test requires NCCL and MPI";
#endif
}
```
