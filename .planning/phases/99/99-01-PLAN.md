### Task 2: Vectorized Kernels (KERN-02)

```cpp
// 128-bit vectorized (float4, int4)
__global__ void quantize_f32_to_int8_vectorized_kernel(
    const float4* src, int4* dst, size_t n4, float scale) {
    size_t idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx >= n4) return;

    float4 vals = src[idx];
    int4 quantized = make_int4(
        __float2int_rn(vals.x / scale),
        __float2int_rn(vals.y / scale),
        __float2int_rn(vals.z / scale),
        __float2int_rn(vals.w / scale)
    );
    dst[idx] = quantized;
}
```

### Task 3: Calibration Histogram Kernel

```cpp
// Shared memory histogram for Phase 100
__global__ void build_histogram_kernel(
    const float* data, uint32_t* histogram,
    size_t n, float min_val, float max_val, int num_bins) {

    __shared__ uint32_t smem[256];

    size_t tid = threadIdx.x;
    if (tid < 256) smem[tid] = 0;
    __syncthreads();

    size_t idx = blockIdx.x * blockDim.x + tid;
    if (idx < n) {
        float val = data[idx];
        int bin = static_cast<int>((val - min_val) / (max_val - min_val) * 255);
        bin = max(0, min(255, bin));
        atomicAdd(&smem[bin], 1);
    }

    __syncthreads();
    if (tid < 256) {
        atomicAdd(&histogram[tid], smem[tid]);
    }
}
```
