**Phase:** 35 — IDE Configuration
**Date:** 2026-04-26
**Status:** ✅ IMPLEMENTED

## Implementation

### 1. Create `.clangd/config.yaml`

```yaml
CompileFlags:
  Add:
    - "-xcuda"
    - "--cuda-gpu-arch=sm_80"
    - "-I/usr/local/cuda/include"
    - "-DNOVA_ENABLE_NCCL=0"
    - "-DNOVA_ENABLE_MPI=0"
  Remove:
    - "-gencode*"
    - "--threads*"
    - "-gencode"
Diagnostics:
  Clang: Off
  Unused: Off
 含糊: Off
```

### 2. Create `.vscode/settings.json`

```json
{
  "clangd.arguments": [
    "--background-index",
    "--compile-commands-dir=${workspaceFolder}"
  ],
  "clangd.path": "clangd",
  "C_Cpp.default.configurationProvider": "ms-vscode.cmake-tools",
  "files.associations": {
    "*.cu": "cuda",
    "*.cuh": "cuda"
  }
}
```

### 3. Create `.vscode/c_cpp_properties.json`

```json
{
  "configurations": [
    {
      "name": "Linux",
      "includePath": [
        "${workspaceFolder}/**",
        "/usr/local/cuda/include"
      ],
      "defines": [],
      "compilerPath": "/usr/local/cuda/bin/nvcc",
      "cStandard": "c17",
      "cppStandard": "c++23",
      "intelliSenseMode": "linux-gcc-x64"
    }
  ],
  "version": 4
}
```

### 4. Create `.vscode/extensions.json`

```json
{
  "recommendations": [
    "llvm-vs-code-extensions.vscode-clangd",
    "ms-vscode.cmake-tools",
    "ms-vscode.cpptools"
  ]
}
```

### 5. Create `docs/ide-setup.md`

Quick guide for setting up clangd and VS Code with Nova.

### 6. Add compile_commands.json symlink to CMakeLists.txt

```cmake
# Create symlink to compile_commands.json
add_custom_command(TARGET cuda_impl POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E create_symlink
        ${CMAKE_BINARY_DIR}/compile_commands.json
        ${CMAKE_SOURCE_DIR}/compile_commands.json
    COMMENT "Creating symlink to compile_commands.json"
)
```
