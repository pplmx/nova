### Task 1: FP8 Type Definitions (FP8-01)

**Files:** `include/cuda/quantize/fp8_types.hpp`

```cpp
namespace nova::quantize {

struct alignas(1) FP8E4M3 {
    uint8_t value;
    
    FP8E4M3() = default;
    explicit FP8E4M3(float f);
    operator float() const;
    
    static constexpr uint8_t POS_INF = 0x7C;
    static constexpr uint8_t NEG_INF = 0xFC;
    static constexpr uint8_t NAN = 0x7D;
    static constexpr uint8_t NEG_ZERO = 0x80;
    
    static constexpr float MAX_NORMAL = 240.0f;
    static constexpr float MIN_NORMAL = 0.015625f;  // 2^-6
};

struct alignas(1) FP8E5M2 {
    uint8_t value;
    
    FP8E5M2() = default;
    explicit FP8E5M2(float f);
    operator float() const;
    
    static constexpr uint8_t POS_INF = 0x7C;
    static constexpr uint8_t NEG_INF = 0xFC;
    static constexpr uint8_t NAN = 0x7D;
    static constexpr uint8_t NEG_ZERO = 0x80;
    
    static constexpr float MAX_NORMAL = 57344.0f;   // 2^15 * 1.75
    static constexpr float MIN_NORMAL = 5.96046e-5f; // 2^-14
};

} // namespace
```

**Conversion Logic (E4M3):**
- Sign: bit 7 (1 = negative)
- Exponent: bits 6-3 (bias = 7, range -6 to 8)
- Mantissa: bits 2-0
- Range: 2^-6 to 240.0

**Conversion Logic (E5M2):**
- Sign: bit 7 (1 = negative)
- Exponent: bits 6-2 (bias = 15, range -14 to 16)
- Mantissa: bits 1-0
- Range: 2^-14 to 57344.0

### Task 2: FP8 Conversion Kernels (FP8-01)

**Files:** `src/cuda/quantize/fp8_kernels.cu`

```cpp
namespace nova::quantize::cuda {

template<typename SrcT, typename DstT>
__global__ void fp8_quantize_kernel(const SrcT* src, DstT* dst, size_t n) {
    size_t idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        dst[idx] = DstT(static_cast<float>(src[idx]));
    }
}

template<typename SrcT, typename DstT>
__global__ void fp8_dequantize_kernel(const SrcT* src, DstT* dst, size_t n, float scale) {
    size_t idx = blockIdx.x * blockDim.x + threadIdx.x;
    if (idx < n) {
        dst[idx] = static_cast<DstT>(static_cast<float>(src[idx])) * scale;
    }
}

void quantize_f32_to_fp8e4m3(const float* src, FP8E4M3* dst, size_t n, cudaStream_t stream = 0);
void quantize_f32_to_fp8e5m2(const float* src, FP8E5M2* dst, size_t n, cudaStream_t stream = 0);
void dequantize_fp8e4m3_to_f32(const FP8E4M3* src, float* dst, size_t n, float scale, cudaStream_t stream = 0);
void dequantize_fp8e5m2_to_f32(const FP8E5M2* src, float* dst, size_t n, float scale, cudaStream_t stream = 0);

} // namespace
```

```cpp
class FP8GEMM {
public:

        const FP8E4M3* a, const FP8E4M3* b,

        float scale_a = 1.0f, float scale_b = 1.0f,

    static void backward(
        const FP8E4M3* grad_output,

        int m, int k, int n,
        cudaStream_t stream = 0
    );
};

} // namespace
```

**Implementation:**
1. Allocate FP32 accumulation buffer
2. For each row i in [0, m):
   - For each col j in [0, n):
     - sum = 0 (FP32)
     - For each k in [0, k):
       - Convert FP8E4M3 to FP32
       - Multiply and accumulate in FP32
     - Store result

### Task 4: FP8 Activations (FP8-03)

**Files:** `include/cuda/quantize/fp8_activation.hpp`

```cpp
namespace nova::quantize {

__device__ __forceinline__ FP8E4M3 fp8_relu(FP8E4M3 x) {
    return (x.value & 0x80) ? FP8E4M3(0.0f) : x;
}

__device__ __forceinline__ FP8E4M3 fp8_gelu(FP8E4M3 x) {
    float f = static_cast<float>(x);
    float cdf = 0.5f * (1.0f + erff(f / sqrtf(2.0f)));
    float result = f * cdf;
    return FP8E4M3(result);
}

void relu_fp8(const FP8E4M3* input, FP8E4M3* output, size_t n, cudaStream_t stream = 0);
void gelu_fp8(const FP8E4M3* input, FP8E4M3* output, size_t n, cudaStream_t stream = 0);

} // namespace
```
