## Implementation

### 1. Create `cmake/NovaConfig.cmake.in`

```cmake
@PACKAGE_INIT@

include("${CMAKE_CURRENT_LIST_DIR}/NovaTargets.cmake")

check_required_components(nova)

# Feature matrix output
if(NOVA_NCCL_ENABLED)
    message(STATUS "Nova @PROJECT_VERSION@: NCCL [ON]")
else()
    message(STATUS "Nova @PROJECT_VERSION@: NCCL [OFF]")
endif()

if(NOVA_MPI_ENABLED)
    message(STATUS "Nova @PROJECT_VERSION@: MPI [ON]")
else()
    message(STATUS "Nova @PROJECT_VERSION@: MPI [OFF]")
endif()

message(STATUS "Nova @PROJECT_VERSION@: CUDA @CMAKE_CUDA_COMPILER_VERSION@")
```

### 2. Update CMakeLists.txt

Add after `project()`:
```cmake
set(NOVA_VERSION ${PROJECT_VERSION})
```

Add before `enable_testing()`:
```cmake
# CMake package export
include(GNUInstallDirs)
install(
    EXPORT NovaTargets 
    FILE NovaTargets.cmake
    NAMESPACE Nova::
    DESTINATION ${CMAKE_INSTALL_LIBDIR}/cmake/nova/${PROJECT_NAME}
)

# Config file
include(CMakePackageConfigHelpers)
configure_package_config_file(
    ${CMAKE_SOURCE_DIR}/cmake/NovaConfig.cmake.in
    ${CMAKE_CURRENT_BINARY_DIR}/NovaConfig.cmake
    INSTALL_DESTINATION ${CMAKE_INSTALL_LIBDIR}/cmake/nova/${PROJECT_NAME}
)
write_basic_package_version_file(
    ${CMAKE_CURRENT_BINARY_DIR}/NovaConfigVersion.cmake
    VERSION ${PROJECT_VERSION}
    COMPATIBILITY SameMajorVersion
)
install(
    FILES 
        ${CMAKE_CURRENT_BINARY_DIR}/NovaConfig.cmake
        ${CMAKE_CURRENT_BINARY_DIR}/NovaConfigVersion.cmake
    DESTINATION ${CMAKE_INSTALL_LIBDIR}/cmake/nova/${PROJECT_NAME}
)
```

### 3. Update library targets

Add `EXPORT NovaTargets` to library definitions:
```cmake
add_library(cuda_impl STATIC ${ALL_CUDA_SOURCES} EXPORT NovaTargets)
```

### 4. Create find_package test

Create `tests/package/CMakeLists.txt`:
```cmake
cmake_minimum_required(VERSION 4.0)
project(NovaPackageTest LANGUAGES CXX CUDA)

find_package(nova REQUIRED)

add_executable(test_link main.cpp)
target_link_libraries(test_link PRIVATE Nova::cuda_impl)
```
